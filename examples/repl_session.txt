# AILANG REPL Session Example
# This demonstrates the new REPL features
# Run with: ailang repl

# Check types of expressions
:type 1 + 2
:type 3.14
:type "hello"
:type true

# Import standard library instances
:import std/prelude

# Show available instances
:instances

# Enable diagnostic output
:dump-core
1 + 2
:dump-core

# Test with type inference
:type 42 == 42

# Enable defaulting trace
:trace-defaulting on
1 + 2
:trace-defaulting off

# Evaluate expressions
1 + 2 * 3
"Hello " ++ "World"
[1, 2, 3]
{x: 10, y: 20}

# Show command history
:history

# Get help
:help

# Exit
:quit
